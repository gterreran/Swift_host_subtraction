import subprocess,sys


[docs]def run(_command):
    pid = subprocess.Popen(_command,shell=True,stdout=subprocess.PIPE,stderr=subprocess.PIPE)
    
    output,error = pid.communicate()
    if error:
        print('ERROR while running '+_command)
        print(error)
        sys.exit()
    return output
